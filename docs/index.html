Quick Start

<script type="module">
  import { Cuss2 } from "https://esm.sh/jsr/@cuss/cuss2-ts@latest";

  const cuss2 = Cuss2.connect("CLIENT_ID", "SECRET");
  await cuss2.connected;
  console.log("Connected to CUSS2 platform!");
</script>
